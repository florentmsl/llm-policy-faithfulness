# Safety-first policy: avoid cars even if it reduces scoring.
# State schema:
# - <Obj>.y is lane index (0 bottom, 10 top).
# - ED(A, B) is distance between objects A and B.
def play(state):
    closest = min(
        state["ED(Chicken1, Car1)"],
        state["ED(Chicken1, Car2)"],
        state["ED(Chicken1, Car3)"],
    )
    if closest < 15:
        return 2  # DOWN (retreat)
    if state["Chicken1.y"] > 0:
        return 2  # keep backing toward the curb
    return 0  # NOOP
