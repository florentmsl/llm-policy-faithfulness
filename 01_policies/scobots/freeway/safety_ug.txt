# Safety-first policy: avoid nearby objects even if it reduces progress.
# State schema:
# - <Obj>.y is lane index (0 bottom, 10 top).
# - ED(A, B) is distance between objects A and B.
def play(state):
    closest = min(
        state["ED(Obj1, Obj3)"],
        state["ED(Obj1, Obj4)"],
        state["ED(Obj1, Obj5)"],
    )
    if closest < 15:
        return 2  # DOWN (retreat)
    if state["Obj1.y"] > 0:
        return 2  # keep backing toward the curb
    return 0  # NOOP
